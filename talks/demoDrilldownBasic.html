DrilldownBasic

import com.knockdata.zeppelin.highcharts._
import com.knockdata.zeppelin.highcharts.model._
import sqlContext.implicits._
val chart = highcharts(bank
  .series("name" -> "marital",
    "y" -> avg(col("balance")))
  .drilldown("name" -> "job",
    "y" -> avg(col("balance"))))
  .chart(Chart.column)

chart.plot()
